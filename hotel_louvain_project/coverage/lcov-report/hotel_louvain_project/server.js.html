const nightsBetween = (start, end) => {
  const a = new Date(start)
  const b = new Date(end)
 
  // normalisation des heures
  a.setHours(0, 0, 0, 0)
  b.setHours(0, 0, 0, 0)
 
  // calcul de la différence en jours
  const diff = b - a
  const nights = Math.ceil(diff / 86400000)
 
  // au minimum une nuit
  return Math.max(1, nights)
}
